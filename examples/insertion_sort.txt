n = 5
array a[5]

a[0] = 3
a[1] = 2
a[2] = 4
a[3] = 1
a[4] = 5

void p() {
    i = 0
    while (i < n) {
        print(a[i])
        i = i + 1
    }
}

void s() {
    i = 1
    while (i < 5) {
        x = a[i]
        j = i - 1
        while (j >= 0 && a[j] > x) {
            a[j + 1] = a[j]
            j = j - 1
        }
        a[j + 1] = x
        i = i + 1
    }
}

p()
s()
p()
